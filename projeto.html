3ï¸âƒ£ LÃ³gica JavaScript

// Elementos do DOM
const inputTarefa = document.getElementById("inputTarefa");
const btnAdicionar = document.getElementById("btnAdicionar");
const listaTarefas = document.getElementById("listaTarefas");

// Array para armazenar tarefas
let tarefas = [];

// Carregar tarefas do localStorage
function carregarTarefas() {
    const dados = localStorage.getItem("minhasTarefas");
    if (dados) {
        tarefas = JSON.parse(dados);
        renderizarTarefas();
    }
}

// Salvar tarefas no localStorage
function salvarTarefas() {
    localStorage.setItem("minhasTarefas", JSON.stringify(tarefas));
}

// Adicionar nova tarefa
function adicionarTarefa() {
    const texto = inputTarefa.value.trim();
    
    if (texto === "") {
        alert("Digite algo!");
        return;
    }
    
    tarefas.push({
        id: Date.now(),
        texto: texto,
        concluida: false
    });
    
    inputTarefa.value = "";
    salvarTarefas();
    renderizarTarefas();
}

// Deletar tarefa
function deletarTarefa(id) {
    tarefas = tarefas.filter(t => t.id !== id);
    salvarTarefas();
    renderizarTarefas();
}

// Marcar como concluÃ­da
function marcarConcluida(id) {
    const tarefa = tarefas.find(t => t.id === id);
    if (tarefa) {
        tarefa.concluida = !tarefa.concluida;
        salvarTarefas();
        renderizarTarefas();
    }
}

// Renderizar tarefas na tela
function renderizarTarefas() {
    listaTarefas.innerHTML = "";
    
    tarefas.forEach(tarefa => {
        const li = document.createElement("li");
        li.className = `tarefa ${tarefa.concluida ? "concluida" : ""}`;
        
        li.innerHTML = `
            <input 
                type="checkbox" 
                class="checkbox"
                ${tarefa.concluida ? "checked" : ""}
                onchange="marcarConcluida(${tarefa.id})"
            >
            <span class="texto">${tarefa.texto}</span>
            <button class="btnDeletar" onclick="deletarTarefa(${tarefa.id})">
                Deletar
            </button>
        `;
        
        listaTarefas.appendChild(li);
    });
}

// Event listeners
btnAdicionar.addEventListener("click", adicionarTarefa);
inputTarefa.addEventListener("keypress", (e) => {
    if (e.key === "Enter") {
        adicionarTarefa();
    }
});

// Carregar tarefas ao abrir a pÃ¡gina
carregarTarefas();
